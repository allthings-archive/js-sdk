# js-sdk [![Build Status](https://travis-ci.org/allthings/js-sdk.svg?branch=master)](https://travis-ci.org/qipp/js-sdk)

## Usage

```js
import API from 'allthings-js-sdk'

const client = API(
  'https://accounts.allthings.me',
  'https://api.allthings.me',
  'MY_CLIENT_ID'
)
```
